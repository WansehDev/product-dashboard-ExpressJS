### 1.2.2. How to run:
Setup Json Settings
``` json
{
  "name": "express-mvc",
  "version": "1.0.0",
  "description": "This is an express mvc framework",
  "main": "app.js",
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1",
    "develop": "nodemon app.js"
  },
  "keywords": [],
  "author": "",
  "license": "ISC",
  "dependencies": {
    "body-parser": "^1.20.0",
    "ejs": "^3.1.6",
    "express": "^4.17.3",
    "mysql": "^2.18.1",
    "path": "^0.12.7"
  }
}

```
Type this in Terminal: 
``` 
npm run develop
```
